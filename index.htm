# Defining a Dice class	
class Dice():
   # Constructor for the Dice class
   def __init__(self,numberOfSides=6):
      
      self.numberOfSides = numberOfSides
      
      self.value = 1

      
   def throw(self):
      
      self.value = random.randint(1,self.numberOfSides)
      
      return self.value

      
# Instantiating a dice object using the Dice class   
dice = Dice(6)
      
dice.throw()
      
print(dice.value)
